// 创建develop分支
git branch develop

// 切换分支
git checkout develop

// 创建并切换分支
git checkout -b develop

// merge 首先切到要合并到的分支上
git merge --no-ff develop

// 推送分支
git push origin develop


// 查看远程分支 -a 全部分支
git branch -r

new-feature

// 删除本地分支
git branch -d feature-new
// 删除远程分支
git push origin --delete feature-new
